An Armstrong number is a number that is equal to the sum of cubes of its digits. For example, 371 is an Armstrong number since 3*3*3 + 7*7*7 + 1*1*1 = 371.

def is_armstrong(n):
    """
    Checks if a given number is an Armstrong number.
    
    Args:
        n (int): The number to check.
    
    Returns:
        bool: True if the number is an Armstrong number, False otherwise.
    """

    # Convert the number to a string to easily access its digits
    str_n = str(n)
    
    # Calculate the number of digits in the number
    num_digits = len(str_n)
    
    # Initialize the sum of cubes of digits
    sum_cubes = 0
    
    # Iterate over each digit in the number
    for digit in str_n:
        # Add the cube of the current digit to the sum
        sum_cubes += int(digit) ** num_digits
    
    # Check if the sum of cubes is equal to the original number
    return n == sum_cubes
